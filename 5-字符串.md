```C++
/*
maxn：状态数
maxm：字符集大小
root：trie图的根
cnt：状态数
init():初始化
insert（）:插入整串
build（）：建
getfail:建fail树
query():查询多少模式串出现
结构体形式直接套外面即可
*/
const int maxn = 5e5 + 10;
const int maxm = 26;
vector<int> f[maxn];
int trie[maxn][maxm];
int fail[maxn];
int ed[maxn];
int root, cnt;
string sp[maxn];
int newnode() {
    for (int i = 0; i < 26; ++i) trie[cnt][i] = -1;
    ed[cnt++] = 0;
    return cnt - 1;
}
void init() {
    cnt = 0;
    root = newnode();
}
void insert(char x[]) {
    int len = strlen(x);
    int now = root;
    for (int i = 0; i < len; ++i) {
        if (trie[now][x[i] - 'a'] == -1) trie[now][x[i] - 'a'] = newnode();
        now = trie[now][x[i] - 'a'];
    }
    ed[now]++;
}
void build() {
    queue<int> que;
    fail[root] = root;
    for (int i = 0; i < 26; ++i) {
        if (trie[root][i] == -1)
            trie[root][i] = root;
        else {
            fail[trie[root][i]] = root;
            que.push(trie[root][i]);
        }
    }
    while (!que.empty()) {
        int now = que.front();
        que.pop();
        for (int i = 0; i < 26; ++i) {
            if (trie[now][i] == -1)
                trie[now][i] = trie[fail[now]][i];
            else {
                fail[trie[now][i]] = trie[fail[now]][i];
                que.push(trie[now][i]);
            }
        }
    }
}

int query(char s[]) {
    int len = strlen(s);
    int now = root;
    int res = 0;
    for (int i = 0; i < len; ++i) {
        now = trie[now][s[i] - 'a'];
        int tmp = now;
        while (tmp != root) {
            res += ed[tmp];
            ed[tmp] = 0;
            tmp = fail[tmp];
        }
    }
    return res;
}
void getfail() {
    for (int i = 1; i < cnt; ++i) f[fail[i]].push_back(i);
}
int main() { return 0; }
```

```C++
        return r[a] == r[b] && r[a + l] == r[b + l];
    }
    void da(int n, int m) {
        n++;
        int p, *x = t1, *y = t2;
        for (int i = 0; i < m; ++i) c[i] = 0;
        for (int i = 0; i < n; ++i) c[x[i] = str[i]]++;
        for (int i = 1; i < m; ++i) c[i] += c[i - 1];
        for (int i = n - 1; i >= 0; --i) sa[--c[x[i]]] = i;
        for (int j = 1; j <= n; j <<= 1) {
            p = 0;
            for (int i = n - j; i < n; ++i) y[p++] = i;
            for (int i = 0; i < n; ++i)
                if (sa[i] >= j) y[p++] = sa[i] - j;
            for (int i = 0; i < m; ++i) c[i] = 0;
            for (int i = 0; i < n; ++i) c[x[y[i]]]++;
            for (int i = 1; i < m; ++i) c[i] += c[i - 1];
            for (int i = n - 1; i >= 0; --i) sa[--c[x[y[i]]]] = y[i];
            swap(x, y);
            p = 1;
            x[sa[0]] = 0;
            for (int i = 1; i < n; ++i)
                x[sa[i]] = cmp(y, sa[i - 1], sa[i], j) ? p - 1 : p++;
            if (p >= n) break;
            m = p;
        }
        int k = 0;
        n--;
        int j;
        for (int i = 0; i <= n; ++i) rk[sa[i]] = i;
        for (int i = 0; i < n; ++i) {
            if (k) k--;
            j = sa[rk[i] - 1];
            while (str[i + k] == str[j + k]) k++;
            he[rk[i]] = k;
        }
    }

        for (int i = 1; i <= n; ++i) rmq[i] = he[i];
        mm[0] = -1;
        for (int i = 1; i <= n; ++i)
            mm[i] = ((i & (i - 1)) == 0) ? mm[i - 1] + 1 : mm[i - 1];
        for (int i = 1; i <= n; ++i) best[0][i] = i;
        for (int i = 1; i <= mm[n]; ++i) {
            for (int j = 1; j + (1 << i) - 1 <= n; ++j) {
                int a = best[i - 1][j];
                int b = best[i - 1][j + (1 << (i - 1))];
                if (rmq[a] < rmq[b])
                    best[i][j] = a;
                else
                    best[i][j] = b;
            }
        }
    }
    int askrmq(int a, int b) {
        int t;
        t = mm[b - a + 1];
        b -= (1 << t) - 1;
        a = best[t][a];
        b = best[t][b];
        return rmq[a] < rmq[b] ? a : b;
    }
    int lcp(int a, int b) {
        a = rk[a];
        b = rk[b];
        if (a > b) swap(a, b);
        return he[askrmq(a + 1, b)];
```

```C++
/*
后缀数组dc3算法:O(n)
sa，rk，he，lcp
数组大小开长度三倍


*/
#define F(x) ((x) / 3 + ((x) % 3 == 1 ? 0 : tb))
#define G(x) ((x) < tb ? (x)*3 + 1 : ((x)-tb) * 3 + 2)
const int maxn = 1e5 + 10;

int wa[maxn * 3], wb[maxn * 3], wv[maxn * 3], wss[maxn * 3];
int c0(int *r, int a, int b) {
    return r[a] == r[b] && r[a + 1] == r[b + 1] && r[a + 2] == r[b + 2];
}
int c12(int k, int *r, int a, int b) {
    if (k == 2)
        return r[a] < r[b] || (r[a] == r[b] && c12(1, r, a + 1, b + 1));
    else
        return r[a] < r[b] || (r[a] == r[b] && wv[a + 1] < wv[b + 1]);
}
void sort(int *r, int *a, int *b, int n, int m) {
    int i;
    for (i = 0; i < n; i++) wv[i] = r[a[i]];
    for (i = 0; i < m; i++) wss[i] = 0;
    for (i = 0; i < n; i++) wss[wv[i]]++;
    for (i = 1; i < m; i++) wss[i] += wss[i - 1];
    for (i = n - 1; i >= 0; i--) b[--wss[wv[i]]] = a[i];
}
void dc3(int *r, int *sa, int n, int m) {
    int i, j, *rn = r + n;
    int *san = sa + n, ta = 0, tb = (n + 1) / 3, tbc = 0, p;
    r[n] = r[n + 1] = 0;
    for (i = 0; i < n; i++)
        if (i % 3 != 0) wa[tbc++] = i;
    sort(r + 2, wa, wb, tbc, m);
    sort(r + 1, wb, wa, tbc, m);
    sort(r, wa, wb, tbc, m);
    for (p = 1, rn[F(wb[0])] = 0, i = 1; i < tbc; i++)
        rn[F(wb[i])] = c0(r, wb[i - 1], wb[i]) ? p - 1 : p++;
    if (p < tbc)
        dc3(rn, san, tbc, p);
    else
        for (i = 0; i < tbc; i++) san[rn[i]] = i;
    for (i = 0; i < tbc; i++)
        if (san[i] < tb) wb[ta++] = san[i] * 3;
    if (n % 3 == 1) wb[ta++] = n - 1;
    sort(r, wb, wa, ta, m);
    for (i = 0; i < tbc; i++) wv[wb[i] = G(san[i])] = i;
    for (i = 0, j = 0, p = 0; i < ta && j < tbc; p++)
        sa[p] = c12(wb[j] % 3, r, wa[i], wb[j]) ? wa[i++] : wb[j++];
    for (; i < ta; p++) sa[p] = wa[i++];
    for (; j < tbc; p++) sa[p] = wb[j++];
}
int sa[maxn * 3], r[maxn];
int rk[maxn], he[maxn];
int s[maxn * 3];
char sp[maxn];
void da(int str[], int sa[], int rk[], int he[], int n, int m) {
```

```C++
/*

ts[]:每个结点的拓扑序

const int maxn = 5e6 + 10;
int len[maxn << 1], link[maxn << 1];
int ch[maxn << 1][26];
int sz, rt, last;
int cnt;
void init() {
    len[1] = link[1] = 0;
    for (int i = 0; i < 26; ++i) ch[1][i] = 0;
    cnt = last = sz = 1;
}
void extend(int c) {
    int p = last;
    int np = ++sz;
    for (int i = 0; i < 26; ++i) ch[np][i] = 0;
    last = np;
    len[np] = len[p] + 1;
    for (; p && !ch[p][c]; p = link[p]) ch[p][c] = np;
    if (!p)
        link[np] = 1;
    else {
        int q = ch[p][c];
        if (len[q] == len[p] + 1) {
            link[np] = q;
        } else {
            int nq = ++sz;
            len[nq] = len[p] + 1;
            memcpy(ch[nq], ch[q], sizeof(ch[q]));
            link[nq] = link[q];
            link[q] = link[np] = nq;
            for (; ch[p][c] == q; p = link[p]) ch[p][c] = nq;
        }
    }
}

int t;
int tc[maxn], ts[maxn << 1];
void sort() {
    memset(tc, 0, sizeof(tc));
    for (int i = 1; i <= sz; ++i) tc[len[i]]++;
    for (int i = 0; i <= sz; ++i) tc[i + 1] += tc[i];
    for (int i = 1; i <= sz; ++i) ts[tc[len[i]]--] = i;
}
```

```C++
    int i = 0, j = nxt[0] = -1;

    int i, j, ans;
    i = j = ans = 0;
    getnxt(x, m);
    while (i < n) {
        while (j != -1 && y[i] != x[j]) j = nxt[j];

        if (j >= m) ans++, j = nxt[j];
    }
```

```C++
char s[maxn], str[maxn << 1];

void Manacher(char s[], int &n) {
    str[0] = '$';
    str[1] = '#';
    for (int i = 0; i < n; ++i)
        str[(i << 1) + 2] = s[i], str[(i << 1) + 3] = '#';
    n = 2 * n + 2;
    str[n] = 0;
    int mx = 0, id;
    for (int i = 1; i < n; ++i) {
        p[i] = mx > i ? min(p[2 * id - i], mx - i) : 1;
        while (str[i - p[i]] == str[i + p[i]]) p[i]++;
        if (p[i] + i > mx) mx = p[i] + i, id = i;
    }
}
int solve(char s[]) {
    int n = strlen(s);
    Manacher(s, n);
    return *max_element(p, p + n) - 1;
}
```

```C++
int nxt[maxn], ext[maxn];
void pre_ekmp(char x[], int m) {
    nxt[0] = m;
    int j = 0;
    while (j + 1 < m && x[j] == x[j + 1]) j++;
    nxt[1] = j;
    int k = 1;
    for (int i = 2; i < m; ++i) {
        int p = nxt[k] + k - 1;
        int L = nxt[i - k];
        if (i + L < p + 1)
            nxt[i] = L;
        else {
            j = max(0, p - i + 1);
            while (i + j < m && x[i + j] == x[j]) j++;
            nxt[i] = j;
            k = i;
        }
    }
}
void ekmp(char x[], int m, char y[], int n) {
    pre_ekmp(x, m);
    int j = 0;
    while (j < n && j < m && x[j] == y[j]) j++;
    ext[0] = j;
    int k = 0;
    for (int i = 1; i < n; ++i) {
        int p = ext[k] + k - 1;
        int L = nxt[i - k];
        if (i + L < p + 1)
            ext[i] = L;
        else {
            j = max(0, p - i + 1);
            while (i + j < n && j < m && y[i + j] == x[j]) j++;
            ext[i] = j, k = i;
        }
    }
}
```

```C++:
const ll seed_pool[] = {146527, 19260817};
const ll mod_pool[] = {1000000009,998244353};
struct Hash{
    ll SEED, MOD;
    vector<ll> p, h;
    Hash(){}
    Hash(const string& s,const int& seed_index,const int& mod_index)
    {
        SEED = seed_pool[seed_index];
        MOD = mod_pool[mod_index];
        int n = s.length();
        p.resize(n + 1), h.resize(n + 1);
        p[0] = 1;

            p[i] = p[i - 1] * SEED % MOD;
        }
        for (int i = 1; i <= n;++i) {
            h[i] = (h[i - 1] * SEED % MOD + s[i - 1]) % MOD;
        }
    }
    ll get(int l, int r) { return (h[r] - h[l] * p[r - 1] % MOD + MOD) % MOD; }
    ll substr(int l, int m) { return get(l, l + m); }
};
```
