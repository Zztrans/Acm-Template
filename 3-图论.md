```c++
struct Edge
{

};
struct MCMF
{
    int n, m, s, t;
    vector<Edge> edges;
    vector<int> G[maxn];
    int inq[maxn];
    ll d[maxn]; //最短路数组
    int p[maxn]; //记录路径
    int a[maxn]; //记录流量
    void init(int n)
    {
        this->n = n;
        for (int i = 0; i < n; i++)
            G[i].clear();
        edges.clear();
    }
    void addedge(int from, int to, int cap, int cost)
    {
        edges.push_back(Edge{from, to, cap, 0, cost});
        edges.push_back(Edge{to, from, 0, 0, -cost});
        m = edges.size();
        G[from].push_back(m - 2);
        G[to].push_back(m - 1);
    }
    bool spfa(int s, int t, int &flow, ll &cost)
    {
        for (int i = 0; i < n; i++)
            d[i] = inf;
        memset(inq, 0, sizeof(inq));
        d[s] = 0;
        inq[s] = 1;
        p[s] = 0;
        a[s] = inf;
        queue<int> q;
        q.push(s);
        while (!q.empty())
        {
            int u = q.front();
            q.pop();
            inq[u] = 0;
            for (int i = 0; i < G[u].size(); i++)
            {
                Edge &e = edges[G[u][i]];
                if (e.cap > e.flow && d[e.to] > d[u] + e.cost)
                {
                    d[e.to] = d[u] + e.cost;             //松弛
                    p[e.to] = G[u][i];                   //记录上一个点
                    a[e.to] = min(a[u], e.cap - e.flow); //流量控制
                    if (!inq[e.to])
                    {
                        q.push(e.to);
                        inq[e.to] = 1;
                    }
                }
            }
        }
        if (d[t] == inf)
            return false; //不存在最短路
        flow += a[t];
        cost += d[t] * a[t];
        int u = t;
        while (u != s)
        {
            edges[p[u]].flow += a[t];
            edges[p[u] ^ 1].flow -= a[t];
            u = edges[p[u]].from;
        }
        return true;
    }
    ll Mincost(int s, int t)
    {
        int flow = 0;ll cost = 0;
        while (spfa(s, t, flow, cost))
            ;
        if(flow==0) cost=-1;
        return cost;
    }
}F;
```

```C++
        G[to].push_back(m - 1);

        memset(vis, 0, sizeof(vis));
        queue<int> q;
        q.push(s);
        d[s] = 0;
        vis[s] = 1;

        {
            int u = q.front();
            q.pop();
            for (int i = 0; i < G[u].size(); i++)
            {
                Edge& e = edges[G[u][i]];
                if (!vis[e.to] && e.cap > e.flow)
                {
                    vis[e.to] = 1;
                    d[e.to] = d[u] + 1;
                    q.push(e.to);
                }
            }
        }
        return vis[t];
    }
    int DFS(int u, int dist)
    {
        if (u == t || dist == 0)
            return dist;
        int flow = 0, f;
        for (int& i = cur[u]; i < G[u].size(); i++)
        {
            Edge& e = edges[G[u][i]];
            if (d[u] + 1 == d[e.to] && (f = DFS(e.to, min(dist, e.cap - e.flow))) > 0)
            {
                e.flow += f;
                edges[G[u][i] ^ 1].flow -= f;
                flow += f;
                dist -= f;
                if (!dist)
                    break;
            }
        }
        return flow;
    }
    int Maxflow(int s, int t)
    {
        this->s = s;
        this->t = t;
        int flow = 0;
        while (BFS())
        {
            memset(cur, 0, sizeof(cur));
            flow += DFS(s, inf);
        }
        return flow;
    }
```

```C++
{
    dfn[u]=low[u]=++num;
```

```C++
int sz[maxn],maxp[maxn],dis[maxn],vis[maxn],q[maxn],rem[maxn];
int sum,rt,ans;
void init() {tot=0; memset(head,-1,sizeof(head)); memset(vis,0,sizeof(vis)); bit.clear();}
void add(int u,int v,int w) {e[tot].to=v; e[tot].nxt=head[u]; e[tot].w=w; head[u]=tot++; }

void getrt(int u,int fa) //找重心
{
    sz[u]=1;maxp[u]=0;
    for(int i=head[u],v;i!=-1;i=e[i].nxt)
    {
        v=e[i].to;
        if(v==fa || vis[v]) continue;
        getrt(v,u);
        sz[u]+=sz[v];
        maxp[u]=max(maxp[u],sz[v]);
    }
    maxp[u]=max(maxp[u],sum-sz[u]);
    if(maxp[u]<maxp[rt]) rt=u; //最大子树sz最小 的为重心
}
void getdis(int u,int fa) //处理距离
{
    rem[++rem[0]]=dis[u];
    for(int i=head[u],v;i!=-1;i=e[i].nxt)
    {
        v=e[i].to;
        if(v==fa || vis[v]) continue;
        dis[v]=dis[u]+e[i].w;
        getdis(v,u);
    }
}
int calc(int u,int v)  //计算以u为根的所有子树的答案
{
    rem[0]=0;
    dis[u]=v;
    getdis(u,0);
    int sum=0;
    sort(rem+1,rem+1+rem[0]);
    int l=1,r=rem[0];
    while(l<r)
    {
        if(rem[l]+rem[r]<=m) sum+=r-l,l++;
        else r--;
    }
    return sum;
}
void solve(int u) //递归计算
{
    vis[u]=1;
    ans+=calc(u,0);
    for(int i=head[u],v;i!=-1;i=e[i].nxt)
    {
        v=e[i].to;
```

```C++
int lca(int x, int y)
{
    if(dep[x] < dep[y]) swap(x, y);
    while(dep[x] > dep[y])
        x = fa[x][lg[dep[x]-dep[y]] - 1];
    if(x == y) return x;
    for(int k = lg[dep[x]] - 1; k >= 0; --k)
        if(fa[x][k] != fa[y][k])
            x = fa[x][k], y = fa[y][k];
    return fa[x][0];
}
```
